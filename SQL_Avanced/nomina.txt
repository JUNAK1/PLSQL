INSERT INTO PAYROLL (EMPLOYEE_ID, PYD, PYD_VALUE) 
SELECT E.EMPLOYEE_ID
     , 1 CONCEPTO
     , (SALARY/30)*NUM_DAYS SALABASICO
FROM EMPLOYEES E, NUM_DAYS D
WHERE E.EMPLOYEE_ID= D.EMPLOYEE_ID;
